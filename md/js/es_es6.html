    # 箭头函数

    (参数1, 参数2) =>{ }

    了解箭头函数的简化写法

    1. this 是静态的，this 始终指向函数声明时所在的作用域下的 this 的值
    2. 不能作为构造函数进行调用
    3. 不能使用 arguments 变量
    4. 箭头函数适合与 this 无关的回调，定时器、数组的方法回调
    5. 箭头函数不适合与 this 有关的回调，事件回调、对象方法

    # Symbol

    Symbol 是 ES6 引用的一种新的原始数据类型，表示独一无二的值。它是 JavaScript 的第七种数据类型，是一种类似于字符串的数据类型

    数据类型：「String、Number、undefined、null、Array、对象（Object）、Symbol」

    # 特点

    1. 它的值是唯一的，用来解决命名冲突问题
    2. 它的值不能与其他数据进行运算
    3. Symbol 类型的对象不能使用 for 循环遍历，但是可以通过 Reflect.ownKeys()来获取对象所有的键名

    # 使用

    1. let s = Symbol('str')，通过该方式创建Symbol对象，虽然传入str值一样，但是返回的不同Symbol对象
    2. let s = Symbol.for('str')，通过该方式创建Symbol对象，只要传入str一样，那么返回的Symbol对象就是相同的
    // 创建 Symbol 对象
let s1 = Symbol()
let s2 = Symbol('abc')
let s3 = Symbol('abc')
let s4 = Symbol.for('abc')
let s5 = Symbol.for('abc')
console.log(s2 === s3)
console.log(s4 === s5)
console.log(s2 === s4)

// 对象添加 Symbol 类型的属性
let person = {
  say: function () {
    console.log('native say...')
  },
}
// 方式1
person[Symbol.for('say')] = function () {
  console.log('symbol say...')
}
person.say()
person[Symbol.for('say')]()
// 方式2
let game = {
  play: function () {
    console.log('playing low game...')
  },
  [Symbol.for('play')]: function () {
    console.log('playing high game...')
  },
}
game.play()
game[Symbol.for('play')]()

    # Promise

    ES6 引入的异步编程的解决方案。语法上 Promise 是一个构造函数，用来封装异步操作并可以获取其成功或失败的结果

    1. let p = new Promise((reslove, reject)=>{ }) // 创建对象
    2. p.then(value=>{ }, reason=>{ }) // 定义回调
    3. p.catch(reason=>{ }) // 语法糖，类似于p.then({}, reason=>{ })
    // 基础语法
const p = new Promise(reslove, reject); // 成功则调用 reslove 方法，失败则调用 reject 方法
p.then(function(value){ }, function(reason){ })// 定义回调函数

    // 异步编程: 读书文件数据
const fs = require('fs')
const { resolve } = require('path')

const p = new Promise(function (resolve, reject) {
  fs.readFile('../每日学习计划.md', (err, data) => {
    // 失败, 调用 reject 方法, 将 p 的结果置为失败
    if (err) reject(err)
    // 成功, 调用 reslove 方法, 将 p 的结果置为成功
    resolve(data.toString())
  })
})

p.then(
  function (value) {
    console.log('success: ' + value)
  },
  function (reason) {
    console.error('faild: ' + reason)
  }
)

    // 异步编程: 发送 ajax 请求
const p = new Promise(function (reslove, reject) {
  const xhr = new XMLHttpRequest()
  xhr.open('GET', 'https://api.apiopen.top/getJoke')
  xhr.send()
  xhr.onload = function () {
    console.log('native success: ' + xhr.responseText)
    reslove(xhr.responseText)
  }
  xhr.onerror = function () {
    console.log('native faild: ')
    reject()
  }
})
p.then(
  function (value) {
    console.log('promise success: ' + value)
  },
  function (reason) {
    console.log('promise faild: ')
  }
)

    # 优点

    1. 将异步调用的代码逻辑和函数回调结果处理的代码逻辑分开，逻辑更清晰
    2. Promise.prototype.then() 方法链式调用，可以解决一系列异步请求的场景

    # Set

    ES6 新定义的数据结构

    /**
 * 集合的应用
 * 1. 数组去重
 * 2. 交集
 * 3. 并集
 * 4. 差集
 */

const a1 = [1, 2, 3, 4, 5, 4, 3, 2, 1]
const a2 = [4, 5, 6, 5, 6]

// 数组去重. 先转为 set 进行元素去重, 然后通过扩展运算符构建数组
let r1 = [...new Set(a1)]
console.log(Array.isArray(r1) + ' : ' + r1)
// 交集
let r2 = [...new Set(a1)].filter((item) => new Set(a2).has(item))
console.log(Array.isArray(r2) + ' : ' + r2)
// 并集
let r3 = [...new Set([...new Set(a1), ...new Set(a2)])]
console.log(Array.isArray(r3) + ' : ' + r3)
// 差集
let r4 = [...new Set(a1)].filter((item) => !new Set(a2).has(item))
console.log(Array.isArray(r4) + ' : ' + r4)

    # 构造方法和普通方法

    1. ES5语法
      1. 构造函数：function Phone(brand, price)
      2. 普通方法：Phone.prototype.call = function(){ }
    2. ES6语法
      1. 构造方法：constructor(brand, price)
      2. 普通方法：call(){ }，方法名()

    # 类的静态成员

    1. ES5语法
      1. 实例对象中不包含函数对象中的属性
      2. 实例对象中不包含函数对象中的方法
      3. 实例对象的属性和方法与函数对象的原型对象prototype是相通的
    2. ES6语法
      1. 实例对象不能访问类变量
      2. 实例对象不能访问类方法
